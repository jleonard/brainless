#animation

A convenience for creating @keyframe animations

#animation > ._(
  @keyframes,
  @time:0.15s,
  @repeat:infinite,
  @easing:linear, 
  @direction: alternate
);
Example
@-webkit-keyframes wiggle {
  0% {-webkit-transform:rotate(5deg);}
  50% {-webkit-transform:rotate(0deg);}
  100% {-webkit-transform:rotate(-5deg);}
}
@-moz-keyframes wiggle {
  0% {-webkit-transform:rotate(5deg);}
  50% {-webkit-transform:rotate(0deg);}
  100% {-webkit-transform:rotate(-5deg);}
}
@-o-keyframes wiggle {
  0% {-webkit-transform:rotate(5deg);}
  50% {-webkit-transform:rotate(0deg);}
  100% {-webkit-transform:rotate(-5deg);}
}
@keyframes wiggle {
  0% {-webkit-transform:rotate(5deg);}
  50% {-webkit-transform:rotate(0deg);}
  100% {-webkit-transform:rotate(-5deg);}
}

.wiggler{
  #animation > ._(
    wiggle,
    0.25s,
    infinite,
    linear,
    alternate
  );
}
